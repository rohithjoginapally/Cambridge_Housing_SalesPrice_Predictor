    function getSearchString() {
        var specialties = getSpecialties();
        var languages = getLanguages();
        var locations = getLocations();
        var facilities = getFacilities();
        var departments = getDepartments();

        var outStr = "query=" + $("#query").val();
        if (specialties.length > 0) {
            for (var i = 0; i < specialties.length; i++) {
                outStr += "&Specialties=" + specialties[i];
            }
        }
        if (languages.length > 0) {
            for (var i = 0; i < languages.length; i++) {
                outStr += "&Languages=" + languages[i];
            }
        }
        if (locations.length > 0) {
            for (var i = 0; i < locations.length; i++) {
                outStr += "&Locations=" + locations[i];
            }
        }
        if (facilities.length > 0) {
            for (var i = 0; i < facilities.length; i++) {
                outStr += "&Facilities=" + facilities[i];
            }
        }
        if (departments.length > 0) {
            for (var i = 0; i < departments.length; i++) {
                outStr += "&Depts=" + departments[i];
            }
        }
        var selectedGenders = $('[name="genderSelect"]:checked'),
            numberSelected = selectedGenders.length;

        if (numberSelected > 0) {
            if (numberSelected == window.totalGenderOptions) {
                outStr += "&Gender=Any";
            } else {
                for (var i = 0; i < numberSelected; i++) {
                    outStr += "&Gender=" + selectedGenders[i].getAttribute("value");
				}
			}
        }

        getDrs();

        return outStr;
    }

    function handleResponse(data, pushState, onResultsLoaded) {
        $('#result-list-wrapper').html(data.SearchResultsHtml);

        $('#bottom-pager').html(data.PagerHtml);

        $('#displaying').html("Displaying page " + (data.PageInfo.PageNumber) + " of " + data.PageInfo.Total + " results");

        if ($("#query").val()) {
            $('#query-display').html('for <strong>' + $("#query").val() + '</strong>');
        } else {
            $('#query-display').html("");
        }

        if (data.SpellCheck.QueryChanged) {
            $("#query-display").html('for <strong>' + data.SpellCheck.RevisedQuery + '</strong>');
            $("#OrigQuery-text").html(data.SpellCheck.OriginalQuery);
            $("#OrigQuery-Link").attr("href", data.SpellCheck.OriginalQuerySearchLink);
            $("#OrigQuery").css('display', 'block');

        } else {
            $("#OrigQuery").css('display', 'none');

        }


		var linkString = $('.pageCountSelector a').attr('href');
        var CntPerPage = 0;
        if (data.PageInfo.NumberPerPage == 10) {
            CntPerPage = 20;
        } else {
            CntPerPage = 10;
        }

        var pageNum = Number(data.PageInfo.PageNumber);
		pageNum = (data.PageInfo.FirstResultShown < CntPerPage) ? 1 : ((data.PageInfo.FirstResultShown - (data.PageInfo.FirstResultShown % CntPerPage)) / CntPerPage) + 1
        linkString = "/profiles/search" + "?" + getSearchString() + (pageNum > 1 ? "&page=" + pageNum : "") + "&count=" + CntPerPage;
        $('.pageCountSelector a').attr('href', linkString) ;

        handleLanguageFacet(data.Facets.Languages);

        // TODO: check when back button data
        if (pushState) {
            var pageState = {};
            pageState.data = data;
            pageState.selections = $('.faculty-wrapper .current-facets').html();
            window.history.pushState(pageState, "Find a Doctor", "/profiles/search" + "?" + getSearchString() + "&page=" + data.PageInfo.PageNumber);
        }

        handleFacilityFacets(data.Facets.Facilities);
        handleSpecialtyFacets(data.Facets.Expertise);
        handleLocationFacets(data.Facets.CityStates);
        handleGenderFacets(data.Facets.Gender);
        handleDepartmentFacets(data.Facets.Departments)

        if (onResultsLoaded !== undefined) {
            onResultsLoaded();
		}
    }


    function handleGenderFacets(facets) {
        var maleResults = false;
        var femaleResults = false;
        for (var i = 0; i < facets.length; i++) {
            if (facets[i].Label == "Male") {
                var maleResults = true;
                var maleCount = facets[i].Count;
            }
            if (facets[i].Label == "Female") {
                var femaleResults = true;
                var femaleCount = facets[i].Count;
            }
        }
        if (!maleResults) {
            $('#genderSelectMale + label').text("Male (0)");
            $('#genderSelectMale').prop('disabled', true);
        } else {
            $('#genderSelectMale + label').text("Male (" + maleCount + ")");
            $('#genderSelectMale').removeAttr('disabled');
        }
        if (!femaleResults) {
            $('#genderSelectFemale + label').text("Female (0)");
            $('#genderSelectFemale').prop('disabled', true);
        } else {
            $('#genderSelectFemale + label').text("Female (" + femaleCount + ")");
            $('#genderSelectFemale').removeAttr('disabled');
        }
    }

    async function getDrs() {
        // get all doctors elements from the list
        let drs = [...document.querySelectorAll(".dr-rev")];

        // create a new array of string ids to make the API call
        let drIds = drs.map(function (d) {
            return Number(d.id)
        })

        // convert list of ids to string to use it in the fetch request and get only what we need
        let drIdsString = drIds.toString()


        if (drIdsString) {

            // fetch request to the api
            const res = await fetch(`https://ratings.md/api/v1/widget/profiles?id=${drIdsString}&account=johns-hopkins-medicine&id_type=npi&pretty`)

            // convert the response to json
            const json = await res.json()

            // api response data
            let data = json.data

            // iterate through the list of doctors and check if the id matches the id from the api to display the ratings
            data.forEach(function (drAPI) {
                // nested iteration to check if the id matches
                drs.forEach(function (dr) {
                    // if the id matches, show the ratings with the following markup and data
                    if (dr.id === String(drAPI.identifier)) {
                        dr.innerHTML = `
                        <div
                            class="ds-summary reputation-summary">
                            <div>
                            <span class="stars ds-stars ds-stars${Math.round(drAPI.rating)}"></span>
                            </div>
                            <div class="xofy average-wrapper">
                            <span class="average">${drAPI.rating}</span>
                            <span class="average-max">out of 5</span>
                            </div>
                            <div class="ratings-comments">
                            <div class="ratings">
                                <span class="rating-count">${drAPI.rating_count} Ratings</span>
                            </div>
                            <div class="comments commentamount-12">
                                <span class="comment-divider">|</span>
                                <span class="comment-count loaded">
                                <span class="nolink-comments">${drAPI.comment_count} Comments</span>
                                </span>
                            </div>
                            </div>
                        </div>
                    `
                    }
                })
            })
        }

    }
